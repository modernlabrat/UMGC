// Comprehensive test with nested case

function main a: integer, b: real returns real;
    c: integer is 8;
    d: real is 7.E2;
begin
    case a is
         when 1 => a * 2 / d ** 2;
         when 2 => a + 5.E+2 - b;
         when 3 =>
             case d is
                 when 1 => a rem 2;
                 others => 9.1E-1;
             endcase;
         when 4 => a / 2 - c;
         others => a + 4.7 * b;
    endcase;
end;
